def _load_mapping(self) -> dict[str, dict]:
    path = pathlib.Path(settings.guidelines_data_path)
    if not path.exists():
        logger.warning(
            "label_to_rny.json not found at %s — all lookups will use training data only",
            path,
        )
        return {}
    return json.loads(path.read_text())
